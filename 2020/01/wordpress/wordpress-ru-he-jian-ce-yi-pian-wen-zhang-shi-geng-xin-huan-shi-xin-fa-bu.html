在百度推送的js代码里，是没有区分文章是更新还是新发布，但百度提供的链接提交的接口里有区分两种情况的。

function check_new_vs_update( $post_id ){
    $myPost        = get_post($post_id);
    $post_created  = new DateTime( $myPost->post_date_gmt );
    $post_modified = new DateTime( $myPost->post_modified_gmt );

    if( abs( $post_created->diff( $post_modified )->s ) <= 1 ){
        // New post
    }else{
        // Updated post
    }
}
add_action('save_post', 'check_new_vs_update' );

完整版代码如下，message会记录在txt文档中：

add_action('save_post', 'wpjam_save_post_notify_baidu_zz', 10, 3);
function wpjam_save_post_notify_baidu_zz($post_id, $post, $update){
   if($post->post_status != 'publish') return;

   $myPost        = get_post($post_id);
   $post_created  = new DateTime( $myPost->post_date_gmt );
   $post_modified = new DateTime( $myPost->post_modified_gmt );

    $msg = 'Is this un update? ';
    $fp =   fopen("error.log","a");
    flock   ($fp, LOCK_EX) ;  

   if( abs( $post_created->diff( $post_modified )->s ) <= 1 ){
         $msg .=  'No.' ;// New post
   }else{
         $msg .= 'Yes.';// Updated post
    }

    fwrite  ($fp, "$msg"."\n");  
    flock   ($fp, LOCK_UN);  
    fclose  ($fp);  
}
